357 float getR10Time(float ms, bool next){
358 static float r10[] = {1,1.25,1.6,2,2.5,3.2,4,5,6.3,8,10};
359 float mult = 1;
360
361 if( ms >= 10000) mult = 10000;
362 else if( ms >= 1000) mult = 1000;
363 else if( ms >= 100) mult = 100;
364 else if( ms >= 10) mult = 10;
365 ms = ms/mult;
366
367 for(int i=9; i>=0;i--){
368 if(ms == r10[i]) return r10[i]* mult;
369 else if(ms > r10[i]){
370 if(next) return r10[i+1] * mult;
371 else return r10[i]* mult;
372 }
373 }
374 return 0;
