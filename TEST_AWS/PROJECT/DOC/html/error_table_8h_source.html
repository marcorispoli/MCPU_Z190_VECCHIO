174 void activateError(QString errcode, unsigned char opt = _ERR_NONE){
175
176 // Find the error in the list of the current active errors
177 if(activeErrors.size()){
178 for(int i=0; i<activeErrors.size(); i++){
179 if(activeErrors[i].key() == errcode){
180 if((opt & _ERR_FORCE) == 0) return;
181 else {
182 if(!(opt & _ERR_NO_SGN)) emit activateErrorSgn(errcode);
183 return;
184 }
185 }
186 }
187 }
188
189 // The error is not in the error list yet
190 errorIterator i = errorList.find(errcode);
191 if(i == errorList.end()) return;
192
193 // Activate the error
194 i.value().setActive(true);
195
196 // AUTOEXPIRE option
197 if(opt & _ERR_AUTOEXPIRE){
198 i.value().setAutoExpire(true);
199 }else {
200 i.value().setAutoExpire(false);
201 }
202
203 // Set the error in the activeError list
204 activeErrors.append(i);
205 if(!(opt & _ERR_NO_SGN)) emit activateErrorSgn(errcode);
206 return;
207 }

224 void clearError(QString errcode, unsigned char opt = _ERR_NONE){
225
226 // Find the error in the active error list
227 for(int i=0; i< activeErrors.size(); i++){
228 if(activeErrors[i].key() == errcode){
229
230 // deactivate the error
231 activeErrors[i].value().setActive(false);
232
233 // Remove the error from the activeErrors
234 activeErrors.removeAt(i);
235 if(!(opt & _ERR_NO_SGN)) emit clearErrorSgn();

251 void clearAutoExpireErrors(unsigned char opt = _ERR_NONE){
252 int i = 0;
253 bool cleared = false;
254
255 // Clear all the errors in the activeErrror list that are flagged as AutoExpire
256 while(i<activeErrors.size()){
